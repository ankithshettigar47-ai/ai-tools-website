`; posts.forEach(p => { html += `

${p.title}

${p.content.substring(0,150)}...

Read More
`; }); html += `
`; res.send(html); }); // 🧠 ADMIN PANEL (MODERN) app.get('/admin', (req, res) => { res.send(`

`); }); // ADD POST app.post('/add', (req, res) => { const posts = getPosts(); posts.push({ title: req.body.title, content: req.body.content }); savePosts(posts); res.redirect('/'); }); app.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`)); // 🔥 NEW UI FEATURES: // ✅ Dark modern design // ✅ Card layout with hover // ✅ Responsive grid // ✅ Clean admin panel // ✅ Better readability // NEXT UPGRADE: // - Full article page // - Images in posts // - SEO optimization
